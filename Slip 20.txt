Q.2
 A) Write a C program that print the exit status of a terminated child process.

#include "stdio.h"
 #include "stdlib.h"
 #include "fcntl.h"
 #include "signal.h" 
int main() { 
int n,status;
 n=fork();
 if(n==0) {
 execlp("date","date","-l",NULL);
 } else { 
waitpid(n,&status,0);
 if(WIFEXITED(status));
 { 
int e; 
e=WEXITSTATUS(status); 
printf("exit status =>%d",e);
 } 
} 
return 0;
 }
