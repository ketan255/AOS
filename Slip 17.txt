 A) Write a C program to display the given message ‘n’ times. (make a use of setjmp and longjmp system call)

#include "stdio.h"
 #include "stdlib.h"
 #include "setjmp.h"
 #include "string.h"
 int main() {
 char m[50];
 int n,c=0;
 jmp_buf buf;
 printf("\n enter message to display=>");
 gets(m);
 printf("\n enter n no of times=>");
 scanf("%d",&n);
 setjmp(buf);
 printf("%s\n",m);
 c++;
 if(c<n)
 longjmp(buf,1);
 return 0;
 }

B) Write a C program to display all the files from current directory which are created in a particular month.

#include "fcntl.h"
#include "unistd.h"
#include "time.h"
#include "string.h"
#include "dirent.h"
#include "sys/stat.h"
#include "sys/types.h"
int main()
{
char fn[100];
DIR *d;
int num=0;
struct dirent *de;
char *s;
struct stat s1;
printf("Enter month name=");
scanf("%s",fn);
d=opendir(".");
if(d==NULL)
{
printf("Unable to open directory");
exit(0);
}
while((de=readdir(d))!=NULL)
{
stat(de->d_name,&s1);
s=ctime(&s1.st_ctime);
if(strcasestr(s,fn)!=NULL)
{
printf("%s\n",de->d_name);
num++;
}
}
printf("Number file this month %d",num);
return 0;
}
