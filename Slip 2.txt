 A) Write a C program that a string as an argument and return all the files that begins with that name in the current directory. For example > ./a.out foo will return all file names that begins with foo

#include "stdio.h"
#include "stdlib.h"
#include "unistd.h"
#include "fcntl.h"
#include "sys/types.h"
#include "sys/stat.h"
#include "dirent.h"
int main(int argc,char *argv[])
{
int c=0;
DIR *d;
d=opendir(".");
struct dirent *dr;
if(d==NULL)
{
printf("unable nto open directory");
exit(0);
}
while((dr=readdir(d))!=NULL)
{
printf("%s\t",dr->d_name);
c++;
}
printf("\n No of files=>%d",c);
closedir(d);
return 0;
}
